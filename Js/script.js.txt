    const slides = document.getElementById("slides");
    const totalSlides = slides.children.length;
    let index = 0;

    function showSlide(i) {
      index = (i + totalSlides) % totalSlides; // wrap around
      slides.style.transform = translateX(-${index * 500}px);
    }

    function nextSlide() {
      showSlide(index + 1);
    }

    function prevSlide() {
      showSlide(index - 1);
    }
